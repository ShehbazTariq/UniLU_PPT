\usepackage{xcolor}
% Syntax: \colorboxed[<color model>]{<color specification>}{<math formula>}
\newcommand*{\colorboxed}{}
\def\colorboxed#1#{%
  \colorboxedAux{#1}%
}
\newcommand*{\colorboxedAux}[3]{%
  % #1: optional argument for color model
  % #2: color specification
  % #3: formula
  \begingroup
    \colorlet{cb@saved}{.}%
    \color#1{#2}%
    \boxed{%
      \color{cb@saved}%
      #3%
    }%
  \endgroup
}

\setbeamercolor{frametitle}{fg=titlenavy}

\setbeamertemplate{frametitle}{%
  \vspace{0.9em}%
  {\usebeamercolor[fg]{frametitle}\usebeamerfont{frametitle}\insertframetitle\par}%

% ── Headline: section label (top-left) + frame number in a corner arc ─────
%  No on-slide navigation bar — jumping around is done via the bottom-centre
%  outline button and the Contents/divider pages.
\setbeamertemplate{headline}{%
  \begin{tikzpicture}[remember picture, overlay]
    % quiet right-aligned counter, inset from the corner
    \node[anchor=north east, text=titlenavy, font=\bfseries\scriptsize]
      at ([xshift=-0.045\paperwidth, yshift=-0.31cm]current page.north east)
      {\insertframenumber/\inserttotalframenumber};
    \draw[arcgrey, line width=0.6pt, line cap=round]
      ([xshift=-0.045\paperwidth, yshift=-0.57cm]current page.north east)
      -- ++(-0.58cm,0);
    % section label, top-left (context only — not a nav control)
    \node[anchor=north west, text=sectiongrey, font=\scriptsize]
      at ([xshift=0.045\paperwidth, yshift=-0.32cm]current page.north west)
      {\ifnum\value{section}>0\hyperlink{sec\thesection}{\secname}\else\secname\fi};
  \end{tikzpicture}%
}

% ── Section divider: a clean "Contents" page before each section ──────────
%  Current section highlighted with its subsections; other sections shaded
%  with their subsections hidden. TOC entries are clickable.
\newcommand{\agendaslide}{%
  \begin{frame}[plain,noframenumbering,t]
    \hypertarget{agenda}{}%
    \begin{tikzpicture}[remember picture, overlay]
      \node[anchor=center, inner sep=0pt]
        at (current page.center)
        {\includegraphics[width=\paperwidth,height=\paperheight]{Assets/outline_bg.png}};
      \node[anchor=south west]
        at ([xshift=0.03\paperwidth, yshift=0.025\paperheight] current page.south west)
        {\includegraphics[height=0.055\paperheight]{Assets/UniLu_short.png}};
      \node[anchor=south east]
        at ([xshift=-0.03\paperwidth, yshift=0.025\paperheight] current page.south east)
        {\includegraphics[height=0.050\paperheight]{Assets/SnT_logo.png}};
    \end{tikzpicture}
    \vspace*{0.32cm}
    {\scriptsize\color{sectiongrey}Overview\par}
    \vspace{0.38cm}
    {\usebeamercolor[fg]{frametitle}\bfseries\Large Agenda\par}
    \vspace{0.12cm}
    {\color{arcgrey}\rule{0.18\paperwidth}{0.7pt}\par}
    \vspace{0.55cm}
    \begin{minipage}{0.82\paperwidth}
      \small
      \setlength{\parskip}{0.30em}
      \setlength{\columnsep}{1.15cm}
      \begin{multicols}{2}
        \tableofcontents[hideallsubsections]
      \end{multicols}
    \end{minipage}
  \end{frame}%
}

\newcommand{\sectionnumberdisplay}{\ifnum\value{section}<10 0\fi\thesection}

\AtBeginSection[]{%
  {%
  \begin{frame}[plain,noframenumbering,t]
    \hypertarget{sec\arabic{section}}{}%
    \begin{tikzpicture}[remember picture, overlay]
      \node[anchor=center, inner sep=0pt]
        at (current page.center)
        {\includegraphics[width=\paperwidth,height=\paperheight]{Assets/outline_bg.png}};
      \fill[titlenavy, opacity=0.03] (current page.south west) rectangle (current page.north east);
      \begin{scope}[shift={([xshift=0.045\paperwidth,yshift=-0.075\paperheight]current page.north west)}]
        \foreach \x in {0,...,10} {
          \foreach \y in {0,...,5} {
            \fill[titlenavy!34] (\x*0.18,-\y*0.18) circle[radius=0.55pt];
          }
        }
      \end{scope}
      \begin{scope}[shift={([xshift=0.865\paperwidth,yshift=-0.535\paperheight]current page.north west)}]
        \foreach \x in {0,...,8} {
          \foreach \y in {0,...,6} {
            \fill[accentblue!68] (\x*0.18,-\y*0.18) circle[radius=0.55pt];
          }
        }
      \end{scope}
      \fill[accentblue!18] ([xshift=0.155\paperwidth,yshift=-0.485\paperheight] current page.north west) circle[radius=0.090\paperwidth];
      \fill[titlenavy] ([xshift=0.155\paperwidth,yshift=-0.485\paperheight] current page.north west) circle[radius=0.069\paperwidth];
      \node[anchor=center, text=white, font=\bfseries\fontsize{34}{36}\selectfont]
        at ([xshift=0.155\paperwidth,yshift=-0.485\paperheight] current page.north west)
        {\sectionnumberdisplay};
      \node[anchor=west, text=titlenavy, font=\bfseries\LARGE, text width=0.56\paperwidth, align=left]
        at ([xshift=0.285\paperwidth, yshift=-0.430\paperheight] current page.north west)
        {\hyperlink{sec\thesection}{\MakeUppercase{\secname}}};
      \node[anchor=north west, text=titlenavy, text width=0.48\paperwidth, align=left]
        at ([xshift=0.287\paperwidth, yshift=-0.500\paperheight] current page.north west)

          \begin{minipage}{0.48\paperwidth}
            \footnotesize
            \setlength{\parskip}{0.15em}
            \tableofcontents[currentsection,sectionstyle=hide,subsectionstyle=show/show/hide]
          \end{minipage}
        };
      \draw[accentred, line width=0.9pt, line cap=round]
        ([xshift=0.285\paperwidth, yshift=-0.470\paperheight] current page.north west)
        -- ++(0.20\paperwidth,0);
      \node[anchor=south west]
        at ([xshift=0.03\paperwidth, yshift=0.025\paperheight] current page.south west)
        {\includegraphics[height=0.055\paperheight]{Assets/UniLu_short.png}};
      \node[anchor=south east]
        at ([xshift=-0.03\paperwidth, yshift=0.025\paperheight] current page.south east)
        {\includegraphics[height=0.050\paperheight]{Assets/SnT_logo.png}};
    \end{tikzpicture}
  \end{frame}
  }%
}
